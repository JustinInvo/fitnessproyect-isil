-- Crear la base de datos BDFitness
CREATE DATABASE BDFitness;

-- Seleccionar la base de datos para usarla
USE BDFitness;

-- Crear la tabla Usuario en la base de datos BDFitness
CREATE TABLE Usuario (
    id INT IDENTITY(1,1) PRIMARY KEY, 
    correo VARCHAR(50) NOT NULL,
    password VARCHAR(255) NOT NULL,
    estado VARCHAR(10) NULL
);

-- Crear la tabla de planes
CREATE TABLE planes (
    id INT PRIMARY KEY,
    nombre TEXT,
    duracion_meses INT,
    entrenamientos_mes INT,
    categoria TEXT,
    precio NUMERIC,
    dias_renovacion INT
);

-- Crear la tabla de clientes
CREATE TABLE clientes (
    id INT PRIMARY KEY,
    nombre TEXT,
    email TEXT,
    telefono TEXT,
    estado_membresia TEXT,
    plan_id INT,
    FOREIGN KEY (plan_id) REFERENCES planes(id)
);

-- Crear la tabla de sanciones
CREATE TABLE sanciones (
    id INT PRIMARY KEY,
    cliente_id INT,
    fecha_inicio DATETIME,
    fecha_fin DATETIME,
    FOREIGN KEY (cliente_id) REFERENCES clientes(id)
);

-- Crear la tabla de trainers
CREATE TABLE trainers (
    id INT PRIMARY KEY,
    nombre TEXT,
    especialidad TEXT
);

-- Crear la tabla de entrenamientos
CREATE TABLE entrenamientos (
    id INT PRIMARY KEY,
    tipo TEXT,
    ubicacion TEXT,
    duracion TEXT,
    max_participantes INT,
    trainer_id INT,
    fecha DATETIME,
    FOREIGN KEY (trainer_id) REFERENCES trainers(id)
);

-- Crear la tabla de reservas
CREATE TABLE reservas (
    id INT PRIMARY KEY,
    cliente_id INT,
    entrenamiento_id INT,
    fecha_reserva DATETIME,
    estado TEXT,
    FOREIGN KEY (cliente_id) REFERENCES clientes(id),
    FOREIGN KEY (entrenamiento_id) REFERENCES entrenamientos(id)
);

-- Crear la tabla de asistencias
CREATE TABLE asistencias (
    id INT PRIMARY KEY,
    reserva_id INT,
    asistencia INT,
    FOREIGN KEY (reserva_id) REFERENCES reservas(id)
);

-- Crear la tabla de reportes
CREATE TABLE reportes (
    id INT PRIMARY KEY,
    tipo TEXT,
    contenido TEXT,
    fecha DATETIME,
    autor_id INT,
    FOREIGN KEY (autor_id) REFERENCES clientes(id)
);
